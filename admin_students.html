/* CSC SAT 2026 — Frontend helpers */ // Auto-dismiss flash alerts after 5 seconds document.addEventListener('DOMContentLoaded', function () { setTimeout(function () { document.querySelectorAll('.alert.fade.show').forEach(function (el) { var bsAlert = bootstrap.Alert.getOrCreateInstance(el); if (bsAlert) bsAlert.close(); }); }, 5000); // Mobile number — digits only var mobileInputs = document.querySelectorAll('input[name="mobile"]'); mobileInputs.forEach(function (inp) { inp.addEventListener('input', function () { this.value = this.value.replace(/\D/g, '').slice(0, 10); }); }); // Name field — letters and spaces only var nameInput = document.querySelector('input[name="name"]'); if (nameInput) { nameInput.addEventListener('input', function () { this.value = this.value.replace(/[^a-zA-Z\s.]/g, ''); }); } });
